69void __rt_thread_scheduler_hook(struct rt_thread *from, struct rt_thread *to)
70{
71 if (NULL != from) {
72 __on_context_switch_out(from->stack_addr);
73 }
74
75 __on_context_switch_in(to->stack_addr);
76}
